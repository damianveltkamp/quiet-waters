# Image Backgrounds (Wallpaper + Widget Spike) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add bundled Christian image backgrounds to the wallpaper builder (shippable on its own), and run a spike to decide the native-widget image path.

**Architecture:** `backgrounds.ts` becomes a discriminated union (`gradient | image`) in one registry so all id-based lookups keep working. A build script converts the source photos to WebP and generates a typed asset module. `WallpaperCanvas` renders image backgrounds under a dark scrim with fixed white text. The widget config screen is restricted to gradients for now; a spike decides whether the native widget can render images before that work is planned.

**Tech Stack:** Expo SDK 57, React Native, TypeScript, `expo-linear-gradient`, `react-native-view-shot`, Zustand, Jest (`jest-expo`) + `@testing-library/react-native`, `sharp` (dev-only, for the asset script).

## Global Constraints

- **Expo SDK 57** — before writing ANY native/widget code, read the pinned docs at `https://docs.expo.dev/versions/v57.0.0/` (per `mobile/AGENTS.md`).
- **iOS-only** app. No Android work.
- **Metro requires literal `require()` paths** — no dynamic/variable require of assets.
- **Bundle target ≈ 6–9 MB** for the image pack: phone-sized WebP at quality ~80.
- **Legibility = fixed white text (`#FFFFFF`) over a dark scrim** — no per-image text-color theming.
- **Images are placeholders.** The current `~/Documents/bgs/` files came from Pinterest with unverified provenance; they are for building against only and must be swapped for clean-owned files before release. Swapping files does not change any code.
- Test runner: `cd mobile && npm test`. Run a single file with `npm test -- <path>`.

Create `mobile/scripts/process-backgrounds.mjs`:
```js
// One-time / regeneration tool. Reads the source photos, writes phone-sized
// WebP into assets/backgrounds/, and generates a typed asset module with a
// literal require() + dominant fallback color per image.
import { readdirSync, mkdirSync, writeFileSync } from 'node:fs';
import { join, extname } from 'node:path';
import { homedir } from 'node:os';
import sharp from 'sharp';

const SRC = process.env.BG_SRC ?? join(homedir(), 'Documents', 'bgs');
const OUT_DIR = join(process.cwd(), 'assets', 'backgrounds');
const GEN = join(process.cwd(), 'src', 'features', 'wallpaper', 'backgroundImages.generated.ts');

const numeric = (name) => parseInt(name.replace(/\D/g, ''), 10) || 0;
const files = readdirSync(SRC)
  .filter((f) => ['.jpg', '.jpeg', '.png'].includes(extname(f).toLowerCase()))
  .sort((a, b) => numeric(a) - numeric(b));

mkdirSync(OUT_DIR, { recursive: true });
const hex = (c) => '#' + [c.r, c.g, c.b].map((n) => n.toString(16).padStart(2, '0')).join('');

const entries = [];
let i = 0;
for (const file of files) {
  i += 1;
  const id = `scene-${String(i).padStart(2, '0')}`;
  const img = sharp(join(SRC, file)).rotate();
  await img
    .clone()
    .resize({ width: 1290, height: 2796, fit: 'cover' })
    .webp({ quality: 80 })
    .toFile(join(OUT_DIR, `${id}.webp`));
  const { dominant } = await img.stats();
  entries.push({ id, fallbackColor: hex(dominant) });
}

const body = entries
  .map((e) => `  { id: '${e.id}', source: require('../../../assets/backgrounds/${e.id}.webp'), fallbackColor: '${e.fallbackColor}' },`)
  .join('\n');

writeFileSync(
  GEN,
  `// AUTO-GENERATED by scripts/process-backgrounds.mjs — do not edit by hand.\n` +
    `import type { ImageSourcePropType } from 'react-native';\n\n` +
    `export interface ImageAsset { id: string; source: ImageSourcePropType; fallbackColor: string }\n\n` +
    `export const IMAGE_ASSETS: readonly ImageAsset[] = [\n${body}\n];\n`,
);

console.log(`Wrote ${entries.length} images to ${OUT_DIR} and ${GEN}`);
```

```ts
function luminance(hex: string): number {
  const n = hex.replace('#', '');
  const rgb = [0, 2, 4].map((i) => parseInt(n.slice(i, i + 2), 16) / 255);
  const lin = rgb.map((c) => (c <= 0.03928 ? c / 12.92 : ((c + 0.055) / 1.055) ** 2.4));
  return 0.2126 * lin[0] + 0.7152 * lin[1] + 0.0722 * lin[2];
}
function contrast(a: string, b: string): number {
  const [hi, lo] = [luminance(a), luminance(b)].sort((x, y) => y - x);
  return (hi + 0.05) / (lo + 0.05);
}
```

```ts
/** @deprecated Prefer `Background`. Kept as an alias for existing imports. */

// Optional friendly display names, keyed by generated id. Fill in as desired;
// unset ids fall back to "Scene N".
```

```tsx
import { Image, ImageBackground, View } from 'react-native';
import { LinearGradient } from 'expo-linear-gradient';
import { ThemedText } from '@/components/ThemedText';
import { spacing } from '@/theme';
import type { Background } from './backgrounds';

interface Props {
  verseText: string;
  reference: string;
  background: Background;
}

export function WallpaperCanvas({ verseText, reference, background }: Props) {
  const content = (
    <>
      <Image
        source={require('../../../assets/images/symbol-white.png')}
        style={{
          width: 22,
          height: 22,
          resizeMode: 'contain',
          tintColor: background.textColor,
          marginBottom: spacing.lg,
        }}
      />
      <View style={{ flexDirection: 'row', alignItems: 'flex-start', justifyContent: 'center' }}>
        <ThemedText variant="quote" align="center" color={background.textColor} style={{ fontSize: 30, lineHeight: 40 }}>
          "{verseText}"
        </ThemedText>
      </View>
      <ThemedText variant="eyebrow" color={background.mutedColor} style={{ marginTop: spacing.lg }}>
        {reference}
      </ThemedText>
    </>
  );

  const centered = {
    flex: 1,
    alignItems: 'center' as const,
    justifyContent: 'center' as const,
    paddingHorizontal: spacing.xl,
  };

  if (background.kind === 'image') {
    const s = background.scrim ?? 0.4;
    const edge = Math.min(s + 0.15, 0.85);
    const mid = s * 0.3;
    const scrim = [`rgba(0,0,0,${edge})`, `rgba(0,0,0,${mid})`, `rgba(0,0,0,${edge})`] as const;
    return (
      <ImageBackground testID="wallpaper-image" source={background.source} resizeMode="cover" style={{ flex: 1 }}>
        <LinearGradient testID="wallpaper-scrim" colors={scrim} style={centered}>
          {content}
        </LinearGradient>
      </ImageBackground>
    );
  }

  return (
    <LinearGradient colors={background.colors} style={centered}>
      {content}
    </LinearGradient>
  );
}
```

```tsx
import { useRouter } from 'expo-router';
import { ThemedText } from '@/components';
import { colors, spacing } from '@/theme';
import { BACKGROUNDS, type Background } from '@/features/wallpaper/backgrounds';
import { useWallpaperDraft } from '@/features/wallpaper/wallpaperDraft';

export default function BackgroundsSheet() {
  const router = useRouter();
  const selectedId = useWallpaperDraft((s) => s.background.id);

            source={bg.source}

          {bg.name}
        </ThemedText>
      </Pressable>

}
```
